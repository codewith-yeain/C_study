for문
	for(초기식; 조건식; 증감식){
		반복할 문장;
	}

--------------------------------------------------------------------
복합 대입 연산자
	+= : num += 값; --> num = num + 값;
	-= : num -= 값; --> num = num - 값;
	*= : num *= 값; --> num = num * 값;
	/= : num /= 값; --> num = num / 값;
	%= : num %= 값; --> num = num % 값;

증감연산자
	num++; --> num += 1; --> num = num + 1;
	num--; --> num -= 1; --> num = num - 1;
